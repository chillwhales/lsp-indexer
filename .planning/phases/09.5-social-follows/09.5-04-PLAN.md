**Create `apps/test/src/components/follower-card.tsx`** — card component for rendering a follow relationship.

Per CONTEXT.md locked decisions:

- **Always visible:** followerAddress, followedAddress (base fields — always present)
- **Conditionally visible:** timestamp (when included)
- **Collapsible section 1:** Follower Profile — shown via `'followerProfile' in obj` guard
- **Collapsible section 2:** Followed Profile — shown via `'followedProfile' in obj` guard
- Full addresses, no truncation (`break-all` CSS wrapping)
- Ghost Button with ChevronDown trigger for collapsible sections
- `'key' in obj` field-presence checks for conditional rendering (DX-04 pattern)

Follow the pattern from `nft-card.tsx` and `owned-token-card.tsx`:

```typescript
'use client';

import { ChevronDown } from 'lucide-react';
import React, { useState } from 'react';

import { Button } from '@/components/ui/button';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Collapsible, CollapsibleContent, CollapsibleTrigger } from '@/components/ui/collapsible';

interface FollowerCardProps {
  follower: Record<string, unknown>;
  index: number;
}

export function FollowerCard({ follower, index }: FollowerCardProps): React.ReactNode {
  const [showFollowerProfile, setShowFollowerProfile] = useState(false);
  const [showFollowedProfile, setShowFollowedProfile] = useState(false);

  return (
    <Card>
      <CardHeader className="pb-3">
        <CardTitle className="text-sm font-medium">Follow #{index + 1}</CardTitle>
      </CardHeader>
      <CardContent className="space-y-3 text-sm">
        {/* Base fields — always present */}
        <div>
          <span className="text-muted-foreground">Follower:</span>
          <p className="font-mono text-xs break-all">{String(follower.followerAddress)}</p>
        </div>
        <div>
          <span className="text-muted-foreground">Followed:</span>
          <p className="font-mono text-xs break-all">{String(follower.followedAddress)}</p>
        </div>

        {/* Conditional scalar fields */}
        {'timestamp' in follower && (
          <div>
            <span className="text-muted-foreground">Timestamp:</span>
            <p className="text-xs">{String(follower.timestamp)}</p>
          </div>
        )}
        {'address' in follower && (
          <div>
            <span className="text-muted-foreground">Contract:</span>
            <p className="font-mono text-xs break-all">{String(follower.address)}</p>
          </div>
        )}

        {/* Collapsible: Follower Profile */}
        {'followerProfile' in follower && follower.followerProfile && (
          <Collapsible open={showFollowerProfile} onOpenChange={setShowFollowerProfile}>
            <CollapsibleTrigger asChild>
              <Button variant="ghost" size="sm" className="w-full justify-start gap-1.5">
                <ChevronDown
                  className={`size-3.5 transition-transform ${showFollowerProfile ? 'rotate-180' : ''}`}
                />
                Follower Profile
              </Button>
            </CollapsibleTrigger>
            <CollapsibleContent className="mt-2 space-y-2 pl-4 border-l">
              {/* Render profile fields from follower.followerProfile */}
              {renderProfileFields(follower.followerProfile as Record<string, unknown>)}
            </CollapsibleContent>
          </Collapsible>
        )}

        {/* Collapsible: Followed Profile */}
        {'followedProfile' in follower && follower.followedProfile && (
          <Collapsible open={showFollowedProfile} onOpenChange={setShowFollowedProfile}>
            <CollapsibleTrigger asChild>
              <Button variant="ghost" size="sm" className="w-full justify-start gap-1.5">
                <ChevronDown
                  className={`size-3.5 transition-transform ${showFollowedProfile ? 'rotate-180' : ''}`}
                />
                Followed Profile
              </Button>
            </CollapsibleTrigger>
            <CollapsibleContent className="mt-2 space-y-2 pl-4 border-l">
              {renderProfileFields(follower.followedProfile as Record<string, unknown>)}
            </CollapsibleContent>
          </Collapsible>
        )}
      </CardContent>
    </Card>
  );
}
```

```typescript
function renderProfileFields(profile: Record<string, unknown>): React.ReactNode {
  return (
    <div className="space-y-1 text-xs">
      {'address' in profile && (
        <div>
          <span className="text-muted-foreground">Address:</span>
          <p className="font-mono break-all">{String(profile.address)}</p>
        </div>
      )}
      {'name' in profile && (
        <div>
          <span className="text-muted-foreground">Name:</span>
          <p>{profile.name ? String(profile.name) : <span className="text-muted-foreground italic">null</span>}</p>
        </div>
      )}
      {'description' in profile && (
        <div>
          <span className="text-muted-foreground">Description:</span>
          <p className="line-clamp-2">{profile.description ? String(profile.description) : <span className="text-muted-foreground italic">null</span>}</p>
        </div>
      )}
      {'followerCount' in profile && (
        <div>
          <span className="text-muted-foreground">Followers:</span>{' '}
          <span>{String(profile.followerCount)}</span>
        </div>
      )}
      {'followingCount' in profile && (
        <div>
          <span className="text-muted-foreground">Following:</span>{' '}
          <span>{String(profile.followingCount)}</span>
        </div>
      )}
      {'tags' in profile && Array.isArray(profile.tags) && profile.tags.length > 0 && (
        <div>
          <span className="text-muted-foreground">Tags:</span>{' '}
          <span>{(profile.tags as string[]).join(', ')}</span>
        </div>
      )}
    </div>
  );
}
```

**Building the include object** — combine scalar toggles with sub-include states:

```typescript
// Inside each tab component
function buildInclude(): FollowerInclude | undefined {
  const anyScalar = scalarIncludes.values.some(([_, v]) => v);
  const anyFollowerProfile = followerProfileIncludes.active;
  const anyFollowedProfile = followedProfileIncludes.active;

  if (!anyScalar && !anyFollowerProfile && !anyFollowedProfile) return undefined;

  const include: FollowerInclude = {};
  for (const [key, value] of scalarIncludes.values) {
    if (value) (include as any)[key] = true;
  }
  if (anyFollowerProfile) {
    include.followerProfile = buildNestedInclude(followerProfileIncludes);
  }
  if (anyFollowedProfile) {
    include.followedProfile = buildNestedInclude(followedProfileIncludes);
  }
  return include;
}
```

Use `useMemo` to memoize the include object to prevent unnecessary re-renders.
