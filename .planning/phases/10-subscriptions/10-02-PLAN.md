    **Pattern for each domain hook:**

    ```typescript
    'use client';

    import type { {Domain}Filter } from '@lsp-indexer/types';
    import type { {Domain} } from '@lsp-indexer/types';
    import {
      build{Domain}Where,
      {domain}Keys,
      parse{Domain}s,
      {Domain}SubscriptionDocument,
    } from '@lsp-indexer/node';
    import type { UseSubscriptionReturn } from '@lsp-indexer/types';
    import { useQueryClient } from '@tanstack/react-query';
    import { useSubscription } from './use-subscription';

    const DEFAULT_LIMIT = 10;

    export interface Use{Domain}SubscriptionParams {
      /** Filter criteria (optional â€” omit for all data) */
      filter?: {Domain}Filter;
      /** Maximum items in subscription result (default: 10) */
      limit?: number;
      /** Enable/disable subscription (default: true) */
      enabled?: boolean;
      /** Invalidate TanStack Query cache on subscription data (default: false) */
      invalidate?: boolean;
      /** Callback when subscription receives new data */
      onData?: (data: {Domain}[]) => void;
      /** Callback when WebSocket reconnects after a drop */
      onReconnect?: () => void;
    }

    export function use{Domain}Subscription(
      params: Use{Domain}SubscriptionParams = {},
    ): UseSubscriptionReturn<{Domain}> {
      const {
        filter,
        limit = DEFAULT_LIMIT,
        enabled = true,
        invalidate = false,
        onData,
        onReconnect,
      } = params;

      const where = build{Domain}Where(filter);

      // Always call useQueryClient unconditionally (Rules of Hooks).
      // Only USE the result when invalidate=true.
      let queryClient: import('@tanstack/react-query').QueryClient | undefined;
      try {
        queryClient = useQueryClient();
      } catch {
        // No QueryClientProvider â€” cache invalidation won't work but hook still functions
      }

      // All 4 type params (TResult, TVariables, TRaw, TParsed) are inferred
      // from the SubscriptionConfig â€” zero explicit type arguments needed.
      return useSubscription(
        {
          document: {Domain}SubscriptionDocument,
          variables: {
            where: Object.keys(where).length > 0 ? where : undefined,
            order_by: {defaultOrderBy},
            limit,
          },
          extract: (result) => result.{hasura_table},
          parser: (raw) => parse{Domain}s(raw),
        },
        {
          enabled,
          invalidate,
          invalidateKeys: invalidate ? [{domain}Keys.all] : undefined,
          queryClient: invalidate ? queryClient : undefined,
          onData,
          onReconnect,
        },
      );
    }
    ```

    **IMPORTANT NOTES:**

    1. **`'use client'` directive** â€” Add to every domain hook file. These are React hooks.

    2. **`useQueryClient()` must be called unconditionally** â€” React's Rules of Hooks forbid conditional hook calls. Always call it, wrapped in try/catch. Only PASS the result when `invalidate=true`.

    3. **`extract` function** â€” This is the key type-safety innovation. Instead of `dataKey: 'universal_profile'` (string that TypeScript can't reason about), use `(result) => result.universal_profile`. TypeScript infers `TRaw` from the return type and validates the parser accepts compatible input.

    4. **`variables` typing** â€” The `variables` object must match the subscription document's variable types exactly. The `where` clause uses the Hasura bool_exp type returned by `buildXWhere`. The `order_by` and `limit` are typed by the document's variables.

    5. **Default sort order** per the locked decision: "fixed newest-first sort (block-order desc for event domains, timestamp desc for entity domains)."
       - **Event domains** (followers, data-changed, token-id-data-changed, universal-receiver): `[{ block_number: 'desc' as const }, { transaction_index: 'desc' as const }, { log_index: 'desc' as const }]`
       - **Entity domains** (profiles, digital-assets, nfts, owned-assets, owned-tokens, creators, issued-assets, encrypted-assets): `undefined` â€” Hasura default ordering

    6. **Parser usage** â€” Call the plural parser without include parameter. Subscription documents include ALL fields by default (all `$includeX: Boolean! = true`), so parsers receive the full type.

    7. **Import pattern** â€” Import domain types from `@lsp-indexer/types`, import functions/documents from `@lsp-indexer/node`, import `useSubscription` from `./use-subscription` (local, same package), import `UseSubscriptionReturn` from `@lsp-indexer/types`.
