**1. Create `NftResult<I>` in `packages/types/src/nfts.ts`:**

NFTs have two nested relations: `collection` (DigitalAsset) and `holder` (NftHolder which is Profile + timestamp). Additionally, NFT has scalar include fields.

Import `IncludeResult` from `'./include-types'`, `DigitalAssetResult` from `'./digital-assets'`, and `ProfileResult` from `'./profiles'`.

NFT base fields: `'address' | 'tokenId' | 'isBurned' | 'isMinted'` (per research — these are identity/boolean fields always present).

```typescript
// Scalar include fields (non-relation)
type NftScalarIncludeFieldMap = {
  formattedTokenId: 'formattedTokenId';
  name: 'name';
  description: 'description';
  category: 'category';
  icons: 'icons';
  images: 'images';
  links: 'links';
  attributes: 'attributes';
};

/**
 * Resolve the collection relation based on include value.
 * - undefined/absent: field absent from type
 * - DigitalAssetInclude object (including {}): narrowed DigitalAsset
 * - Treat any truthy object as sub-include
 */
type ResolveNftCollection<I> = I extends { collection: infer C }
  ? C extends DigitalAssetInclude
    ? { collection: DigitalAssetResult<C> | null }
    : {}
  : {};

/**
 * Resolve the holder relation based on include value.
 * NftHolder = Profile + { timestamp: string }. When holder is narrowed,
 * we narrow the profile fields but always keep `timestamp` + `address` as base.
 */
type ResolveNftHolder<I> = I extends { holder: infer H }
  ? H extends ProfileInclude
    ? { holder: (ProfileResult<H> & { timestamp: string }) | null }
    : {}
  : {};

/**
 * NFT type narrowed by include parameter.
 *
 * Base fields (always present): address, tokenId, isBurned, isMinted
 * Scalar fields: narrowed by include map
 * Relations: collection (DigitalAsset), holder (NftHolder) — narrowed recursively
 */
export type NftResult<I extends NftInclude | undefined = undefined> = I extends undefined
  ? Nft
  : IncludeResult<
      Nft,
      'address' | 'tokenId' | 'isBurned' | 'isMinted',
      NftScalarIncludeFieldMap,
      I
    > &
      ResolveNftCollection<NonNullable<I>> &
      ResolveNftHolder<NonNullable<I>>;
```

Export `NftResult`.
