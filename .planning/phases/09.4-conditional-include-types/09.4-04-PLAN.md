**1. Create `OwnedTokenResult<I>` in `packages/types/src/owned-tokens.ts`:**

OwnedTokens have 4 nested relations, each with its own include type:

- `digitalAsset` → `DigitalAssetInclude` → uses `DigitalAssetResult<C>`
- `nft` → `OwnedTokenNftInclude` (NftInclude minus collection/holder) → needs custom narrowing
- `ownedAsset` → `OwnedTokenOwnedAssetInclude` (3 scalar fields: balance, block, timestamp) → needs custom narrowing
- `holder` → `ProfileInclude` → uses `ProfileResult<H>`

Import `IncludeResult` from `'./include-types'`, `DigitalAssetResult` from `'./digital-assets'`, `ProfileResult` from `'./profiles'`.

OwnedToken base fields: `'id' | 'digitalAssetAddress' | 'holderAddress' | 'tokenId'`

```typescript
// Scalar include fields for OwnedToken
type OwnedTokenScalarIncludeFieldMap = {
  block: 'block';
  timestamp: 'timestamp';
};

/**
 * Resolve nested digitalAsset relation.
 * Uses DigitalAssetResult for narrowing — same as OwnedAsset.
 */
type ResolveOwnedTokenDA<I> = I extends { digitalAsset: infer C }
  ? C extends DigitalAssetInclude
    ? { digitalAsset: DigitalAssetResult<C> | null }
    : {}
  : {};

/**
 * Resolve nested nft relation.
 * OwnedTokenNftInclude has 8 fields (NftInclude minus collection/holder).
 * Since the NFT in owned-token context doesn't have collection/holder relations,
 * we build a specific narrowed type.
 */
type OwnedTokenNftScalarFieldMap = {
  formattedTokenId: 'formattedTokenId';
  name: 'name';
  description: 'description';
  category: 'category';
  icons: 'icons';
  images: 'images';
  links: 'links';
  attributes: 'attributes';
};

type ResolveOwnedTokenNft<I> = I extends { nft: infer N }
  ? N extends OwnedTokenNftInclude
    ? {
        nft: IncludeResult<
          Nft,
          'address' | 'tokenId' | 'isBurned' | 'isMinted',
          OwnedTokenNftScalarFieldMap,
          N
        > | null;
      }
    : {}
  : {};

/**
 * Resolve nested ownedAsset relation.
 * OwnedTokenOwnedAssetInclude has 3 fields: balance, block, timestamp.
 * OwnedAsset base fields in this context: id, digitalAssetAddress, holderAddress.
 */
type OwnedTokenOwnedAssetFieldMap = {
  balance: 'balance';
  block: 'block';
  timestamp: 'timestamp';
};

type ResolveOwnedTokenOA<I> = I extends { ownedAsset: infer O }
  ? O extends OwnedTokenOwnedAssetInclude
    ? {
        ownedAsset: IncludeResult<
          OwnedAsset,
          'id' | 'digitalAssetAddress' | 'holderAddress',
          OwnedTokenOwnedAssetFieldMap,
          O
        > | null;
      }
    : {}
  : {};

/**
 * Resolve nested holder relation (Profile).
 */
type ResolveOwnedTokenHolder<I> = I extends { holder: infer H }
  ? H extends ProfileInclude
    ? { holder: ProfileResult<H> | null }
    : {}
  : {};

/**
 * OwnedToken type narrowed by include parameter.
 *
 * Base fields (always present): id, digitalAssetAddress, holderAddress, tokenId
 * Scalar fields: block, timestamp
 * Relations: digitalAsset, nft (minus collection/holder), ownedAsset (3 scalars), holder
 */
export type OwnedTokenResult<I extends OwnedTokenInclude | undefined = undefined> =
  I extends undefined
    ? OwnedToken
    : IncludeResult<
        OwnedToken,
        'id' | 'digitalAssetAddress' | 'holderAddress' | 'tokenId',
        OwnedTokenScalarIncludeFieldMap,
        I
      > &
        ResolveOwnedTokenDA<NonNullable<I>> &
        ResolveOwnedTokenNft<NonNullable<I>> &
        ResolveOwnedTokenOA<NonNullable<I>> &
        ResolveOwnedTokenHolder<NonNullable<I>>;
```
