**Step 2: Create `apps/test/src/components/nft-card.tsx`**

Extract the NftCard as a separate component (following established pattern from digital-asset-card.tsx and profile-card.tsx).

The card shows:

- Token ID (and formatted token ID if available)
- Collection name + symbol
- Owner address (truncated) + acquisition timestamp
- First metadata icon (or fallback placeholder)
- Description snippet (if available)
- Burn/mint status badges
- Attribute count badge

```typescript
'use client';

import type { Nft } from '@lsp-indexer/types';
import { Badge } from '@/components/ui/badge';
import { Card, CardContent, CardDescription, CardHeader, CardTitle } from '@/components/ui/card';
import { Gem } from 'lucide-react'; // or similar NFT icon

interface NftCardProps {
  nft: Nft;
}

export function NftCard({ nft }: NftCardProps) {
  const iconUrl = nft.icons?.[0]?.url;

  return (
    <Card>
      <CardHeader>
        <div className="flex items-center gap-3">
          <div className="size-10 rounded-lg bg-muted flex items-center justify-center overflow-hidden shrink-0">
            {iconUrl ? (
              <img src={iconUrl} alt="" className="size-full object-cover" />
            ) : (
              <Gem className="size-5 text-muted-foreground" />
            )}
          </div>
          <div className="min-w-0 flex-1">
            <CardTitle className="text-base flex items-center gap-2">
              <span className="truncate">
                {nft.collectionName ?? 'Unknown Collection'}
              </span>
              {nft.collectionSymbol && (
                <span className="text-muted-foreground text-sm">({nft.collectionSymbol})</span>
              )}
            </CardTitle>
            <CardDescription className="font-mono text-xs truncate">
              Token: {nft.formattedTokenId ?? nft.tokenId}
            </CardDescription>
          </div>
          <div className="flex gap-1">
            {nft.isBurned && (
              <Badge variant="destructive" className="text-xs">Burned</Badge>
            )}
            {!nft.isMinted && (
              <Badge variant="outline" className="text-xs">Not Minted</Badge>
            )}
          </div>
        </div>
      </CardHeader>
      <CardContent className="space-y-2">
        {nft.description && (
          <p className="text-sm text-muted-foreground line-clamp-2">{nft.description}</p>
        )}
        <div className="flex flex-wrap gap-4 text-sm text-muted-foreground">
          {nft.owner && (
            <span className="font-mono text-xs">
              Owner: {nft.owner.address.slice(0, 6)}...{nft.owner.address.slice(-4)}
            </span>
          )}
          {nft.attributes && nft.attributes.length > 0 && (
            <span>{nft.attributes.length} traits</span>
          )}
        </div>
      </CardContent>
    </Card>
  );
}
```

3. **`buildNftFilter(debouncedValues)` function:**

```typescript
function buildNftFilter(debouncedValues: Record<string, string>): NftFilter | undefined {
  const f: NftFilter = {};
  if (debouncedValues.collectionAddress) f.collectionAddress = debouncedValues.collectionAddress;
  if (debouncedValues.tokenId) f.tokenId = debouncedValues.tokenId;
  if (debouncedValues.ownerAddress) f.ownerAddress = debouncedValues.ownerAddress;
  if (debouncedValues.isBurned === 'true') f.isBurned = true;
  if (debouncedValues.isBurned === 'false') f.isBurned = false;
  if (debouncedValues.isMinted === 'true') f.isMinted = true;
  if (debouncedValues.isMinted === 'false') f.isMinted = false;
  return Object.keys(f).length > 0 ? f : undefined;
}
```
