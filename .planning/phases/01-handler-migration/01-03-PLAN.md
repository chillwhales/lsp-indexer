```
/**
 * Decimals entity handler (post-verification).
 *
 * Reads the `decimals()` value for newly discovered Digital Assets via
 * Multicall3. Runs in Step 5.5 (post-verification) because it needs the
 * list of newly verified Digital Assets from the verification phase.
 *
 * Behavior:
 *   - postVerification: true — runs in Step 5.5, not Step 3
 *   - Reads batchCtx.getVerified(EntityCategory.DigitalAsset).newEntities
 *   - Batches decimals() calls via Multicall3.aggregate3Static (100 per batch)
 *   - Creates Decimals entities and adds to BatchContext for pipeline persistence
 *   - Queues enrichment for digitalAsset FK
 *   - Handles Multicall3 failures gracefully (some assets may not have decimals)
 *
 * Port from v1: packages/indexer/src/app/handlers/decimalsHandler.ts
 */
```

Implementation:

```typescript
const BATCH_SIZE = 100;
const ENTITY_TYPE = 'Decimals';

const DecimalsHandler: EntityHandler = {
  name: 'decimals',
  listensToBag: ['LSP7Transfer', 'LSP8Transfer', 'DataChanged'],
  postVerification: true,

  async handle(hctx: HandlerContext, _triggeredBy: string): Promise<void> {
    const { context, batchCtx } = hctx;

    // Read newly verified Digital Assets from the verification phase
    const newDAs = batchCtx.getVerified(EntityCategory.DigitalAsset).newEntities;
    if (newDAs.size === 0) return;

    const newDAsList = [...newDAs.values()];
    const batchesCount = Math.ceil(newDAsList.length / BATCH_SIZE);

    for (let index = 0; index < batchesCount; index++) {
      const start = index * BATCH_SIZE;
      const batch = newDAsList.slice(start, start + BATCH_SIZE);

      let results: Aggregate3StaticReturn | undefined;
      try {
        results = await aggregate3StaticLatest(
          context,
          batch.map((da) => ({
            target: da.id,
            allowFailure: true,
            callData: LSP7DigitalAsset.functions.decimals.encode({}),
          })),
        );
      } catch (error) {
        context.log.warn(
          JSON.stringify({
            message: 'Multicall3 batch failed for decimals, skipping batch',
            batchIndex: index,
            batchSize: batch.length,
            error: error instanceof Error ? error.message : String(error),
          }),
        );
        continue;
      }

      results.forEach((result, i) => {
        const da = batch[i];
        if (result.success && isHex(result.returnData) && result.returnData !== '0x') {
          try {
            const entity = new Decimals({
              id: da.id,
              address: da.id,
              digitalAsset: null, // FK initially null — resolved by enrichment
              value: hexToNumber(result.returnData),
            });
            batchCtx.addEntity(ENTITY_TYPE, entity.id, entity);
            batchCtx.queueEnrichment<Decimals>({
              category: EntityCategory.DigitalAsset,
              address: da.id,
              entityType: ENTITY_TYPE,
              entityId: entity.id,
              fkField: 'digitalAsset',
            });
          } catch (error) {
            context.log.warn(
              JSON.stringify({
                message: 'Failed to parse decimals value',
                address: da.id,
                error: error instanceof Error ? error.message : String(error),
              }),
            );
          }
        }
      });
    }
  },
};

export default DecimalsHandler;
```

```
/**
 * FormattedTokenId entity handler.
 *
 * Populates `NFT.formattedTokenId` based on the LSP8TokenIdFormat for the
 * contract. Handles two scenarios:
 *
 * 1. New NFTs created in this batch — looks up the current LSP8TokenIdFormat
 *    (from BatchContext or DB) and formats the token ID.
 * 2. LSP8TokenIdFormat changes — retroactively reformats ALL existing NFTs
 *    for the affected contract by querying the DB.
 *
 * Declares dependsOn: ['lsp8TokenIdFormat'] to ensure the format data is
 * available in BatchContext before this handler runs.
 *
 * When the format is unknown or not yet set, formattedTokenId is left as null.
 * Logs a warning for unknown format values.
 *
 * Port from v1: packages/indexer/src/utils/entityPopulation.ts:339-416
 */
```

Implementation:

```typescript
import { formatTokenId } from '@/utils';
import { EntityHandler, HandlerContext } from '@/core/types';
import { LSP8TokenIdFormat, NFT } from '@chillwhales/typeorm';
import { In, Not } from 'typeorm';
import { isHex } from 'viem';

const NFT_ENTITY_TYPE = 'NFT';

const FormattedTokenIdHandler: EntityHandler = {
  name: 'formattedTokenId',
  listensToBag: ['LSP8Transfer', 'DataChanged'],
  dependsOn: ['lsp8TokenIdFormat'],

  async handle(hctx: HandlerContext, triggeredBy: string): Promise<void> {
    const { store, batchCtx } = hctx;

    // Collect NFTs from BatchContext (created by NFT handler from LSP8Transfer events)
    const batchNfts = batchCtx.getEntities<NFT>(NFT_ENTITY_TYPE);

    // Collect LSP8TokenIdFormat entities from BatchContext (created by lsp8TokenIdFormat handler)
    const batchFormats = batchCtx.getEntities<LSP8TokenIdFormat>('LSP8TokenIdFormat');

    // Track which NFT IDs we've already updated (to avoid double-processing in Path 2)
    const updatedNftIds = new Set<string>();

    // -----------------------------------------------------------------------
    // Path 1: New NFTs in this batch — look up format and set formattedTokenId
    // Mutate the NFT entity objects IN PLACE (they're Map references from BatchContext).
    // Do NOT call addEntity() — the 'NFT' bag key may be sealed as a raw type.
    // -----------------------------------------------------------------------
    if (batchNfts.size > 0) {
      // Get unique contract addresses from new NFTs
      const nftAddresses = [...new Set([...batchNfts.values()].map((nft) => nft.address))];

      // Load existing formats from DB for contracts not in batch
      const dbFormats = await store.findBy(LSP8TokenIdFormat, {
        id: In(nftAddresses),
      });
      const dbFormatMap = new Map(dbFormats.map((f) => [f.id, f]));

      for (const [id, nft] of batchNfts) {
        // Batch format takes precedence over DB format
        const format = batchFormats.get(nft.address) ?? dbFormatMap.get(nft.address);

        if (!isHex(nft.tokenId)) {
          nft.formattedTokenId = null;
        } else if (!format || format.value === null) {
          // Format unknown or not set — leave null per user decision
          nft.formattedTokenId = null;
        } else {
          const formatted = formatTokenId({
            tokenId: nft.tokenId,
            lsp8TokenIdFormat: format.value,
          });
          if (formatted === null) {
            console.warn(
              `[formattedTokenId] Unknown format for ${nft.address}: ${String(format.value)}`,
            );
          }
          nft.formattedTokenId = formatted;
        }

        // Entity mutated in place — no addEntity() call needed
        updatedNftIds.add(id);
      }
    }

    // -----------------------------------------------------------------------
    // Path 2: Format changes — retroactively update ALL existing NFTs
    // -----------------------------------------------------------------------
    if (batchFormats.size > 0) {
      const formatAddresses = [...batchFormats.keys()];

      // Query ALL NFTs for affected contracts from DB, excluding ones we already updated
      const existingNfts = await store.findBy(NFT, {
        address: In(formatAddresses),
        ...(updatedNftIds.size > 0 ? { id: Not(In([...updatedNftIds])) } : {}),
      });

      if (existingNfts.length > 0) {
        for (const nft of existingNfts) {
          const format = batchFormats.get(nft.address);
          if (!format || format.value === null) {
            nft.formattedTokenId = null;
          } else if (!isHex(nft.tokenId)) {
            nft.formattedTokenId = null;
          } else {
            const formatted = formatTokenId({
              tokenId: nft.tokenId,
              lsp8TokenIdFormat: format.value,
            });
            if (formatted === null) {
              console.warn(
                `[formattedTokenId] Unknown format for ${nft.address}: ${String(format.value)}`,
              );
            }
            nft.formattedTokenId = formatted;
          }
        }

        // Upsert retroactively updated NFTs directly via store
        // These are existing DB entities, not BatchContext entities, so we upsert them directly.
        // This is acceptable because FormattedTokenId is updating a single field on existing entities.
        await store.upsert(existingNfts);
      }
    }
  },
};

export default FormattedTokenIdHandler;
```

IMPORTANT NOTES:

- `listensToBag: ['LSP8Transfer', 'DataChanged']` — triggered by both new NFTs (LSP8Transfer) and format changes (DataChanged which triggers lsp8TokenIdFormat handler)
- `dependsOn: ['lsp8TokenIdFormat']` — ensures format entities are in BatchContext before this handler runs
- Path 1 reads from BatchContext NFT bag and mutates entity objects IN PLACE (the Map references are the same objects). Do NOT call `addEntity()` for the 'NFT' bag key — it may be sealed as a raw type from Step 2.
- Path 2 queries DB for existing NFTs when format changes — this is the retroactive update. Uses `store.upsert()` directly since these entities aren't in BatchContext.
