function parseAndAddSubEntities(
  entity: LSP3Profile,
  data: unknown,
  hctx: HandlerContext,
): { success: true } | { success: false; fetchErrorMessage: string } {
  if (typeof data !== 'object' || data === null) {
    return { success: false, fetchErrorMessage: 'Error: Invalid data' };
  }

  const typed = data as LSP3ProfileMetadataJSON;
  if (!typed.LSP3Profile) {
    return { success: false, fetchErrorMessage: 'Error: Invalid LSP3Profile' };
  }

  const { name, description, tags, links, avatar, profileImage, backgroundImage } =
    typed.LSP3Profile;

  const parentRef = new LSP3Profile({ id: entity.id });

  // Name
  const nameEntity = new LSP3ProfileName({
    id: uuidv4(),
    lsp3Profile: parentRef,
    value: name,
  });
  hctx.batchCtx.addEntity(SUB_TYPES.NAME, nameEntity.id, nameEntity);

  // Description
  const descEntity = new LSP3ProfileDescription({
    id: uuidv4(),
    lsp3Profile: parentRef,
    value: description,
  });
  hctx.batchCtx.addEntity(SUB_TYPES.DESCRIPTION, descEntity.id, descEntity);

  // Tags (array)
  if (tags && Array.isArray(tags)) {
    for (const tag of tags) {
      const tagEntity = new LSP3ProfileTag({
        id: uuidv4(),
        lsp3Profile: parentRef,
        value: tag,
      });
      hctx.batchCtx.addEntity(SUB_TYPES.TAG, tagEntity.id, tagEntity);
    }
  }

  // Links (array of {title, url})
  if (links && Array.isArray(links)) {
    for (const { title, url } of links) {
      const linkEntity = new LSP3ProfileLink({
        id: uuidv4(),
        lsp3Profile: parentRef,
        title,
        url,
      });
      hctx.batchCtx.addEntity(SUB_TYPES.LINK, linkEntity.id, linkEntity);
    }
  }

  // Assets (avatar array, filtered by isFileAsset)
  if (avatar && Array.isArray(avatar)) {
    for (const asset of avatar.filter(isFileAsset)) {
      const assetEntity = new LSP3ProfileAsset({
        id: uuidv4(),
        lsp3Profile: parentRef,
        url: asset.url,
        fileType: asset.fileType,
        ...(asset.verification &&
          isVerification(asset.verification) && {
            verificationMethod: asset.verification.method,
            verificationData: asset.verification.data,
            verificationSource: (asset.verification as { source?: string }).source,
          }),
      });
      hctx.batchCtx.addEntity(SUB_TYPES.ASSET, assetEntity.id, assetEntity);
    }
  }

  // Profile Images (flat array, not nested like LSP4)
  if (profileImage && Array.isArray(profileImage)) {
    for (const img of profileImage) {
      const imgEntity = new LSP3ProfileImage({
        id: uuidv4(),
        lsp3Profile: parentRef,
        url: img.url,
        width: img.width,
        height: img.height,
        ...(img.verification &&
          isVerification(img.verification) && {
            verificationMethod: img.verification.method,
            verificationData: img.verification.data,
            verificationSource: (img.verification as { source?: string }).source,
          }),
      });
      hctx.batchCtx.addEntity(SUB_TYPES.IMAGE, imgEntity.id, imgEntity);
    }
  }

  // Background Images (flat array)
  if (backgroundImage && Array.isArray(backgroundImage)) {
    for (const img of backgroundImage) {
      const bgEntity = new LSP3ProfileBackgroundImage({
        id: uuidv4(),
        lsp3Profile: parentRef,
        url: img.url,
        width: img.width,
        height: img.height,
        ...(img.verification &&
          isVerification(img.verification) && {
            verificationMethod: img.verification.method,
            verificationData: img.verification.data,
            verificationSource: (img.verification as { source?: string }).source,
          }),
      });
      hctx.batchCtx.addEntity(SUB_TYPES.BACKGROUND_IMAGE, bgEntity.id, bgEntity);
    }
  }

  return { success: true };
}

**CRITICAL FK CHAIN ISSUE:** The `LSP29AccessControlCondition` entity has FK `encryption` pointing to `LSP29EncryptedAssetEncryption`, NOT to `LSP29EncryptedAsset`. The shared utility's `queueClear` uses parent entity IDs based on the main entity. For the access control conditions, we need to handle this differently:

Two approaches:

1. **Rely on DB cascade deletes** if the LSP29EncryptedAssetEncryption → LSP29AccessControlCondition relationship has `onDelete: CASCADE` in the TypeORM entity definition. Check the entity schema first.
2. **Queue a separate clear** for AccessControlCondition using encryption entity IDs. But we don't know encryption IDs at this point (they're in the DB, not in the batch).

**Recommended approach:** The V1 handler `clearSubEntities()` does a 2-step query: (1) find encryptions by encrypted asset ID, (2) find conditions by encryption IDs, (3) delete conditions, (4) delete other sub-entities. For V2, the simplest path is:

- Include all sub-entity types EXCEPT `LSP29AccessControlCondition` in `subEntityDescriptors`
- Handle `LSP29AccessControlCondition` clearing as part of the `parseAndAddSubEntities` callback — after confirming the clear mechanism works

Actually, check the schema: if `LSP29EncryptedAssetEncryption` has `onDelete: CASCADE` for its relationship with `LSP29AccessControlCondition`, then clearing encryptions automatically cascades to conditions. In that case, only the 6 direct sub-entities need `queueClear`.

If no cascade: add a custom clear step in the handler's `handle()` method, before calling `handleMetadataFetch()`, that queries existing encryptions for the affected entity IDs and queues condition clears.

**Best practice for this plan:** Do NOT rely on cascades. Explicitly handle the FK chain by NOT including `LSP29AccessControlCondition` in the shared `subEntityDescriptors`. Instead, override the success path in the handler to queue an additional clear for conditions. Or better: add the clear inside `parseAndAddSubEntities` where we have access to `hctx.batchCtx.queueClear`.

function parseAndAddSubEntities(
  entity: LSP29EncryptedAsset,
  data: unknown,
  hctx: HandlerContext,
):
  | { success: true; entityUpdates: Record<string, unknown> }
  | { success: false; fetchErrorMessage: string } {
  if (typeof data !== 'object' || data === null) {
    return { success: false, fetchErrorMessage: 'Error: Invalid data' };
  }

  const typed = data as LSP29EncryptedAssetJSON;
  if (!typed.LSP29EncryptedAsset) {
    return { success: false, fetchErrorMessage: 'Error: Invalid LSP29EncryptedAsset' };
  }

  const {
    version,
    id: contentId,
    title,
    description,
    images,
    revision,
    createdAt,
    file,
    encryption,
    chunks,
  } = typed.LSP29EncryptedAsset;

  const parentRef = new LSP29EncryptedAsset({ id: entity.id });

  // Title
  const titleEntity = new LSP29EncryptedAssetTitle({
    id: uuidv4(),
    lsp29EncryptedAsset: parentRef,
    value: title,
  });
  hctx.batchCtx.addEntity('LSP29EncryptedAssetTitle', titleEntity.id, titleEntity);

  // Description
  const descEntity = new LSP29EncryptedAssetDescription({
    id: uuidv4(),
    lsp29EncryptedAsset: parentRef,
    value: description,
  });
  hctx.batchCtx.addEntity('LSP29EncryptedAssetDescription', descEntity.id, descEntity);

  // File (optional)
  if (file) {
    const fileEntity = new LSP29EncryptedAssetFile({
      id: uuidv4(),
      lsp29EncryptedAsset: parentRef,
      type: file.type,
      name: file.name,
      size: file.size !== undefined ? BigInt(file.size) : null,
      lastModified: file.lastModified !== undefined ? BigInt(file.lastModified) : null,
      hash: file.hash,
    });
    hctx.batchCtx.addEntity('LSP29EncryptedAssetFile', fileEntity.id, fileEntity);
  }

  // Encryption (optional)
  let encryptionEntity: LSP29EncryptedAssetEncryption | null = null;
  if (encryption) {
    encryptionEntity = new LSP29EncryptedAssetEncryption({
      id: uuidv4(),
      lsp29EncryptedAsset: parentRef,
      method: encryption.method,
      ciphertext: encryption.ciphertext,
      dataToEncryptHash: encryption.dataToEncryptHash,
      decryptionCode: encryption.decryptionCode,
      decryptionParams: encryption.decryptionParams
        ? JSON.stringify(encryption.decryptionParams)
        : null,
    });
    hctx.batchCtx.addEntity('LSP29EncryptedAssetEncryption', encryptionEntity.id, encryptionEntity);

    // Access Control Conditions (nested under encryption)
    if (encryption.accessControlConditions && Array.isArray(encryption.accessControlConditions)) {
      encryption.accessControlConditions.forEach((condition, index) => {
        const condEntity = new LSP29AccessControlCondition({
          id: uuidv4(),
          encryption: encryptionEntity!,
          conditionIndex: index,
          contractAddress: condition.contractAddress,
          chain: condition.chain,
          method: condition.method,
          standardContractType: condition.standardContractType,
          comparator: condition.comparator || condition.returnValueTest?.comparator,
          value: condition.returnValueTest?.value,
          tokenId:

              ? condition.parameters.find((p) => p.startsWith('0x') && p.length === 66) || null
              : null,
          followerAddress:
            condition.method === 'isFollowing' && condition.parameters
              ? condition.parameters[0]
              : null,
          rawCondition: JSON.stringify(condition),
        });
        hctx.batchCtx.addEntity('LSP29AccessControlCondition', condEntity.id, condEntity);
      });
    }
  }

  // Chunks (optional)
  if (chunks) {
    const chunksEntity = new LSP29EncryptedAssetChunks({
      id: uuidv4(),
      lsp29EncryptedAsset: parentRef,
      cids: chunks.cids || [],
      iv: chunks.iv,
      totalSize: chunks.totalSize !== undefined ? BigInt(chunks.totalSize) : null,
    });
    hctx.batchCtx.addEntity('LSP29EncryptedAssetChunks', chunksEntity.id, chunksEntity);
  }

  // Images (array of arrays, filtered by isFileImage)
  if (images && Array.isArray(images)) {
    images
      .filter((imageSet) => Array.isArray(imageSet))
      .forEach((imageSet, index) => {
        imageSet.filter(isFileImage).forEach(({ url, width, height, verification }) => {
          const imgEntity = new LSP29EncryptedAssetImage({
            id: uuidv4(),
            lsp29EncryptedAsset: parentRef,
            url,
            width,
            height,
            ...(verification &&
              isVerification(verification) && {
                verificationMethod: verification.method,
                verificationData: verification.data,
                verificationSource: (verification as { source?: string }).source,
              }),
            imageIndex: index,
          });
          hctx.batchCtx.addEntity('LSP29EncryptedAssetImage', imgEntity.id, imgEntity);
        });
      });
  }

  // Return entity-level metadata updates (version, contentId, revision, createdAt)
  return {
    success: true,
    entityUpdates: {
      version,
      contentId,
      revision,
      createdAt: createdAt ? new Date(createdAt) : null,
    },
  };
}
