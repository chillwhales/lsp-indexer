```typescript
// Entity comparison types
interface EntityDefinition {
  name: string; // e.g., 'UniversalProfile'
  hasuraTable: string; // e.g., 'universal_profile' (snake_case)
  primaryKey: string; // usually 'id'
  category: 'core' | 'event' | 'metadata' | 'ownership' | 'lsp' | 'custom';
  isMetadataSub: boolean; // true for ProfileImage, AssetImage, Tags, etc.
}

interface KnownDivergence {
  entityType: string;
  field: string;
  reason: string;
}

interface ComparisonConfig {
  v1Url: string;
  v2Url: string;
  v1Secret?: string;
  v2Secret?: string;
  entities?: string[]; // filter to specific entity types
  sampleSize: number; // rows to sample per entity (default 100)
}

interface CountResult {
  entityName: string;
  v1Count: number;
  v2Count: number;
  match: boolean;
}

interface FieldDiff {
  field: string;
  v1Value: unknown;
  v2Value: unknown;
}

interface RowDiff {
  entityName: string;
  rowId: string;
  diffs: FieldDiff[];
  knownDivergences: FieldDiff[]; // diffs that match known exclusions
  unexpectedDiffs: FieldDiff[]; // diffs that are bugs
}

interface ComparisonReport {
  counts: CountResult[];
  sampleDiffs: RowDiff[];
  missingEntityTypes: { endpoint: 'v1' | 'v2'; entityName: string }[];
  passed: boolean;
}
```

Create `packages/indexer-v2/src/comparison/graphqlClient.ts` that provides a Hasura GraphQL client using the `axios` dependency already in package.json (no new dependencies needed).

Implement `createGraphqlClient(url: string, adminSecret?: string)` factory function returning an object with:

1. **`queryCount(hasuraTable: string): Promise<number>`** — Sends a GraphQL query:

   ```graphql
   query { {hasuraTable}_aggregate { aggregate { count } } }
   ```

   Returns the count number. Handles errors gracefully (return -1 if table doesn't exist, which means the entity type is missing from this endpoint).

2. **`querySampleIds(hasuraTable: string, limit: number): Promise<string[]>`** — Fetches a sample of row IDs:

   ```graphql
   query { {hasuraTable}(limit: {limit}, order_by: { id: asc }) { id } }
   ```

   Returns array of ID strings.

3. **`queryRowsByIds(hasuraTable: string, ids: string[]): Promise<Record<string, unknown>[]>`** — Fetches full rows by IDs:

   ```graphql
   query { {hasuraTable}(where: { id: { _in: [{ids}] } }, order_by: { id: asc }) { ...all_fields } }
   ```

   Since we don't know the fields upfront, use Hasura's introspection to discover fields. Implement a `queryTableFields(hasuraTable: string): Promise<string[]>` method that does:

   ```graphql
   query {
     __type(name: "{hasuraTable}") {
       fields {
         name
         type {
           name
           kind
         }
       }
     }
   }
   ```

   Cache the result per table. Only include scalar fields (not object/array relations) — filter by `kind === 'SCALAR'` or name types like `String`, `Int`, `Boolean`, `bigint`, `timestamptz`, etc. Exclude `__typename`.

4. **`checkHealth(): Promise<boolean>`** — Simple health check that runs `query { __typename }`.

Important implementation details:

- Use `axios.post()` with `{ query: string, variables?: Record<string, unknown> }` body
- Set headers: `Content-Type: application/json`, and `x-hasura-admin-secret: {secret}` if adminSecret provided
- Wrap all requests in try/catch — log errors but don't crash
- For querySampleIds, if the table has no `id` column (shouldn't happen but safety), return empty array
- Cache introspected field lists in a `Map<string, string[]>` to avoid repeated introspection

Follow conventions: factory function pattern (createGraphqlClient), single quotes, no console.log (use a simple status tracking approach — the CLI in Plan 02 will handle output).
