**Step 2: Create `packages/node/src/parsers/data-key-resolver.ts`**

This utility builds a reverse map (hex → name) from all LUKSO data key constants and exposes a lookup function.

```typescript
/**
 * ERC725Y data key name resolver.
 *
 * Builds a reverse map from hex data key → human-readable name using official
 * LUKSO contract packages. The map is built once at module load time.
 *
 * @example
 * resolveDataKeyName('0x5ef83ad9559033e6e941db7d7c495acdce616347d28e90c7ce47cbfcfcad3bc5')
 * // → 'LSP3Profile'
 *
 * resolveDataKeyName('0xdeadbeef...')
 * // → null (unknown key)
 */
```

**Import the DataKeys objects** from each package:

```typescript
import { LSP8DataKeys } from '@lukso/lsp8-contracts';

import { LSP10DataKeys } from '@lukso/lsp10-contracts';
import { LSP12DataKeys } from '@lukso/lsp12-contracts';
import { LSP17DataKeys } from '@lukso/lsp17contractextension-contracts';
```

**Build the reverse map:**

```typescript
const ALL_DATA_KEY_SOURCES: Record<string, Record<string, string>> = {
  LSP1: LSP1DataKeys,
  LSP3: LSP3DataKeys,
  LSP4: LSP4DataKeys,
  LSP5: LSP5DataKeys,
  LSP6: LSP6DataKeys,
  LSP8: LSP8DataKeys,
  LSP9: LSP9DataKeys,
  LSP10: LSP10DataKeys,
  LSP12: LSP12DataKeys,
  LSP17: LSP17DataKeys,
};

// Build reverse map: lowercase hex → human-readable name
const DATA_KEY_MAP = new Map<string, string>();

for (const [_prefix, keys] of Object.entries(ALL_DATA_KEY_SOURCES)) {
  for (const [name, hexValue] of Object.entries(keys)) {
    if (typeof hexValue === 'string' && hexValue.startsWith('0x')) {
      // Store lowercase for case-insensitive lookup
      DATA_KEY_MAP.set(hexValue.toLowerCase(), name);
    }
  }
}
```

**Export the lookup function:**

```typescript
/**
 * Resolve a raw hex ERC725Y data key to its human-readable name.
 *
 * Uses a pre-built reverse map from all known LUKSO LSP data keys.
 * Returns null for unknown keys.
 *
 * @param hexKey - Raw hex data key (e.g., '0x5ef83ad9...')
 * @returns Human-readable name (e.g., 'LSP3Profile') or null
 */
export function resolveDataKeyName(hexKey: string): string | null {
  return DATA_KEY_MAP.get(hexKey.toLowerCase()) ?? null;
}
```

**NOTES:**

- The map is built at module load time (lazy initialization not needed — it's a small constant map)
- All hex values stored lowercase for case-insensitive matching (Hasura may return mixed case)
- Some data keys are "array keys" with dynamic parts (e.g., `LSP5ReceivedAssets[]` has a fixed first 16 bytes + variable index). For now, only exact-match keys are resolved. Array/mapping key resolution can be added later.

**3-overload `<const I>` pattern (DX-05):**

```typescript
export function parseDataChangedEvent(raw: any): DataChangedEvent;
export function parseDataChangedEvent<const I extends DataChangedEventInclude>(
  raw: any,
  include: I,
): DataChangedEventResult<I>;
export function parseDataChangedEvent(
  raw: any,
  include?: DataChangedEventInclude,
): DataChangedEvent | PartialDataChangedEvent {
  const result: DataChangedEvent = {
    // Base fields (always present)
    address: raw.address,
    dataKey: raw.data_key,
    dataValue: raw.data_value,
    dataKeyName: resolveDataKeyName(raw.data_key),

    // Includable scalars
    blockNumber: raw.block_number ?? null,
    timestamp: raw.timestamp ?? null,
    logIndex: raw.log_index ?? null,
    transactionIndex: raw.transaction_index ?? null,

    // Relations — delegate to existing parsers with sub-include
    universalProfile: raw.universalProfile
      ? parseProfile(raw.universalProfile, include?.universalProfile)
      : null,
    digitalAsset: raw.digitalAsset
      ? parseDigitalAsset(raw.digitalAsset, include?.digitalAsset)
      : null,
  };

  if (include) {
    return stripExcluded(result, include, [
      'address',
      'dataKey',
      'dataValue',
      'dataKeyName',
    ]) as DataChangedEventResult<typeof include>;
  }
  return result;
}

export function parseDataChangedEvents(raw: any[]): DataChangedEvent[];
export function parseDataChangedEvents<const I extends DataChangedEventInclude>(
  raw: any[],
  include: I,
): DataChangedEventResult<I>[];
export function parseDataChangedEvents(
  raw: any[],
  include?: DataChangedEventInclude,
): (DataChangedEvent | PartialDataChangedEvent)[] {
  if (include) return raw.map((r) => parseDataChangedEvent(r, include));
  return raw.map((r) => parseDataChangedEvent(r));
}
```

**NOTES:**

- `raw.data_key` → `dataKey` — Hasura field is `data_key`, domain type is `dataKey`
- `raw.data_value` → `dataValue` — same pattern
- `raw.block_number` → `blockNumber` — Int!, always present in Hasura but null in our type when not included
- `parseProfile` and `parseDigitalAsset` handle their own sub-include narrowing recursively
- `include?.universalProfile` can be `true` (boolean) or `ProfileInclude` (object) — both profile parsers accept this
- `include?.digitalAsset` can be `true` or `DigitalAssetInclude` — DA parser accepts this via the `as any` cast pattern
- **The `as any` cast pattern for sub-include:** When passing `include?.universalProfile` to `parseProfile`, if it's `true` (boolean), parseProfile receives `true` as the include param. Check how existing parsers handle `true` vs object include. If `parseProfile` only accepts `ProfileInclude | undefined` (not `boolean`), pass `undefined` when `true` and the actual include object when it's an object:

```typescript
const upInclude = include?.universalProfile;
const profileInclude = typeof upInclude === 'object' && upInclude != null ? upInclude : undefined;
universalProfile: raw.universalProfile
```

```typescript
export function buildDataChangedEventIncludeVars(
  include?: DataChangedEventInclude,
): Record<string, boolean> {
  if (!include) return {};

  const activeUP = hasActiveIncludes(include.universalProfile);
  const activeDA = hasActiveIncludes(include.digitalAsset);

  const vars: Record<string, boolean> = {
    includeBlockNumber: include.blockNumber ?? false,
    includeTimestamp: include.timestamp ?? false,
    includeLogIndex: include.logIndex ?? false,
    includeTransactionIndex: include.transactionIndex ?? false,
    includeUniversalProfile: activeUP,
    includeDigitalAsset: activeDA,
  };

  // UP sub-includes with "UniversalProfile" prefix (matching encrypted-assets pattern)
  if (activeUP) {
    const profileVars = buildProfileIncludeVars(include.universalProfile);
    for (const [key, val] of Object.entries(profileVars)) {
      vars[key.replace('includeProfile', 'includeUniversalProfile')] = val;
    }
  }

  // DA sub-includes with "DigitalAsset" prefix (matching issued-assets pattern)
  if (activeDA) {
    const daVars = buildDigitalAssetIncludeVars(include.digitalAsset);
    for (const [key, val] of Object.entries(daVars)) {
      vars[key.replace('include', 'includeDigitalAsset')] = val;
    }
  }

  return vars;
}
```
