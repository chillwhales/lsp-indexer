Per CONTEXT.md locked decisions:

- **Top-level row (always visible):** `creatorAddress`, `digitalAssetAddress` (base fields — always present)

Follow the pattern from `follower-card.tsx` (closest pattern — two collapsible relation sections):

```typescript
'use client';

import { ChevronDown } from 'lucide-react';
import React, { useState } from 'react';

import { Button } from '@/components/ui/button';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Collapsible, CollapsibleContent, CollapsibleTrigger } from '@/components/ui/collapsible';
import { RawJsonToggle } from '@/components/playground';

interface CreatorCardProps {
  creator: Record<string, unknown>;
  index: number;
}

export function CreatorCard({ creator, index }: CreatorCardProps): React.ReactNode {
  const [showCreatorProfile, setShowCreatorProfile] = useState(false);
  const [showDigitalAsset, setShowDigitalAsset] = useState(false);

  return (
    <Card>
      <CardHeader className="pb-3">
        <CardTitle className="text-sm font-medium">Creator #{index + 1}</CardTitle>
      </CardHeader>
      <CardContent className="space-y-3 text-sm">
        {/* Base fields — always present */}
        <div>
          <span className="text-muted-foreground">Creator:</span>
          <p className="font-mono text-xs break-all">{String(creator.creatorAddress)}</p>
        </div>
        <div>
          <span className="text-muted-foreground">Digital Asset:</span>
          <p className="font-mono text-xs break-all">{String(creator.digitalAssetAddress)}</p>
        </div>

        {/* Conditional scalar fields */}
        {'arrayIndex' in creator && (
          <div>
            <span className="text-muted-foreground">Array Index:</span>
            <p className="text-xs">{creator.arrayIndex != null ? String(creator.arrayIndex) : <span className="text-muted-foreground italic">null</span>}</p>
          </div>
        )}
        {'interfaceId' in creator && (
          <div>
            <span className="text-muted-foreground">Interface ID:</span>
            <p className="font-mono text-xs">{creator.interfaceId ? String(creator.interfaceId) : <span className="text-muted-foreground italic">null</span>}</p>
          </div>
        )}
        {'timestamp' in creator && (
          <div>
            <span className="text-muted-foreground">Timestamp:</span>
            <p className="text-xs">{String(creator.timestamp)}</p>
          </div>
        )}

        {/* Collapsible: Creator Profile */}
        {'creatorProfile' in creator && creator.creatorProfile && (
          <Collapsible open={showCreatorProfile} onOpenChange={setShowCreatorProfile}>
            <CollapsibleTrigger asChild>
              <Button variant="ghost" size="sm" className="w-full justify-start gap-1.5">
                <ChevronDown
                  className={`size-3.5 transition-transform ${showCreatorProfile ? 'rotate-180' : ''}`}
                />
                Creator Profile
              </Button>
            </CollapsibleTrigger>
            <CollapsibleContent className="mt-2 space-y-2 pl-4 border-l">
              {renderProfileFields(creator.creatorProfile as Record<string, unknown>)}
            </CollapsibleContent>
          </Collapsible>
        )}

        {/* Collapsible: Digital Asset */}
        {'digitalAsset' in creator && creator.digitalAsset && (
          <Collapsible open={showDigitalAsset} onOpenChange={setShowDigitalAsset}>
            <CollapsibleTrigger asChild>
              <Button variant="ghost" size="sm" className="w-full justify-start gap-1.5">
                <ChevronDown
                  className={`size-3.5 transition-transform ${showDigitalAsset ? 'rotate-180' : ''}`}
                />
                Digital Asset
              </Button>
            </CollapsibleTrigger>
            <CollapsibleContent className="mt-2 space-y-2 pl-4 border-l">
              {renderDigitalAssetFields(creator.digitalAsset as Record<string, unknown>)}
            </CollapsibleContent>
          </Collapsible>
        )}

        <RawJsonToggle data={creator} />
      </CardContent>
    </Card>
  );
}
```

```typescript
function ListTab({ mode }: { mode: HookMode }): React.ReactNode {
  const [address, setAddress] = useState(''); // optional pre-filter
  const filterFields = useFilterFields(FILTERS);
  const [sortField, setSortField] = useState<CreatorSortField | ''>('');
  const [sortDirection, setSortDirection] = useState<SortDirection>('desc');
  const [sortNulls, setSortNulls] = useState<SortNulls | ''>('');
  const [limit, setLimit] = useState(10);
  const [offset, setOffset] = useState(0);

  // Include toggles for scalar fields + sub-includes for relations
  const scalarIncludes = useIncludeToggles(CREATOR_INCLUDE_FIELDS);
  const creatorProfileIncludes = useSubInclude(PROFILE_INCLUDE_FIELDS);
  const digitalAssetIncludes = useSubInclude(DIGITAL_ASSET_INCLUDE_FIELDS);

  // Build include object — combine scalar toggles with sub-include states
  // (same pattern as follows page, see buildInclude helper)
  const include: CreatorInclude | undefined = useMemo(() => {
    const anyScalar = scalarIncludes.values.some(([_, v]) => v);
    const anyProfile = creatorProfileIncludes.active;
    const anyDA = digitalAssetIncludes.active;

    if (!anyScalar && !anyProfile && !anyDA) return undefined;

    const inc: CreatorInclude = {};
    for (const [key, value] of scalarIncludes.values) {
      if (value) (inc as any)[key] = true;
    }
    if (anyProfile) {
      inc.creatorProfile = buildNestedInclude(creatorProfileIncludes);
    }
    if (anyDA) {
      inc.digitalAsset = buildNestedInclude(digitalAssetIncludes);
    }
    return inc;
  }, [scalarIncludes.values, creatorProfileIncludes, digitalAssetIncludes]);

  // Build filter and sort
  const filter: CreatorFilter = { ...filterFields.values };
  const sort: CreatorSort | undefined = sortField
    ? { field: sortField as CreatorSortField, direction: sortDirection, ...(sortNulls ? { nulls: sortNulls as SortNulls } : {}) }
    : undefined;

  // Select hook based on mode
  const useCreatorsHook = mode === 'client' ? useCreatorsReact : useCreatorsNext;
  const result = useCreatorsHook({
    filter: Object.values(filter).some(Boolean) ? filter : undefined,
    sort,
    limit,
    offset,
    ...(include ? { include } : {}),
  });

  return (
    <div className="space-y-4">
      {/* Filters */}
      <FilterFieldsRow fields={FILTERS} {...filterFields} />

      {/* Sort */}
      <SortControls options={SORTS} ... />

      {/* Include toggles — scalar + two sub-include sections */}
      <IncludeToggles fields={CREATOR_INCLUDE_FIELDS} {...scalarIncludes} />
      <SubIncludeSection title="Creator Profile" fields={PROFILE_INCLUDE_FIELDS} {...creatorProfileIncludes} />
      <SubIncludeSection title="Digital Asset" fields={DIGITAL_ASSET_INCLUDE_FIELDS} {...digitalAssetIncludes} />

      {/* Pagination controls */}
      ...

      {/* Results */}
      <ResultsList
        items={result.creators}
        isLoading={result.isLoading}
        error={result.error}
        renderItem={(creator, i) => (
          <CreatorCard creator={creator as Record<string, unknown>} index={i} />
        )}
      />

      {/* Total count */}
      ...
    </div>
  );
}
```
