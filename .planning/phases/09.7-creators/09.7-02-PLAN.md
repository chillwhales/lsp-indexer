```typescript
import type { Creator, CreatorInclude, CreatorResult, PartialCreator } from '@lsp-indexer/types';
import { parseDigitalAsset } from './digital-assets';
import { parseProfile } from './profiles';
import { stripExcluded } from './strip';

/**
 * Parse a Hasura lsp4_creator response into a clean Creator type.
 *
 * Maps: creator_address → creatorAddress, address → digitalAssetAddress,
 * array_index (numeric) → arrayIndex (number | null), interface_id → interfaceId.
 * Delegates profile/DA parsing to existing parsers with sub-includes.
 *
 * @overload No include param — returns full Creator type
 * @overload With include param — returns narrowed CreatorResult<I>
 */
export function parseCreator(raw: any): Creator;
export function parseCreator<const I extends CreatorInclude>(
  raw: any,
  include: I,
): CreatorResult<I>;
export function parseCreator(raw: any, include?: CreatorInclude): Creator | PartialCreator {
  const result: Creator = {
    creatorAddress: raw.creator_address,
    digitalAssetAddress: raw.address,
    arrayIndex: raw.array_index != null ? Number(raw.array_index) : null,
    interfaceId: raw.interface_id ?? null,
    timestamp: raw.timestamp,
    creatorProfile: raw.creatorProfile
      ? parseProfile(raw.creatorProfile, include?.creatorProfile)
      : null,
    digitalAsset: raw.digitalAsset
      ? parseDigitalAsset(raw.digitalAsset, include?.digitalAsset)
      : null,
  };

  if (include) {
    return stripExcluded(result, include, [
      'creatorAddress',
      'digitalAssetAddress',
    ]) as CreatorResult<typeof include>;
  }
  return result;
}

/**
 * Parse an array of Hasura lsp4_creator responses.
 */
export function parseCreators(raw: any[]): Creator[];
export function parseCreators<const I extends CreatorInclude>(
  raw: any[],
  include: I,
): CreatorResult<I>[];
export function parseCreators(raw: any[], include?: CreatorInclude): (Creator | PartialCreator)[] {
  if (include) return raw.map((r) => parseCreator(r, include));
  return raw.map((r) => parseCreator(r));
}
```

This matches the established pattern in follower parser (which calls `parseProfile(raw, include?.followerProfile)`) and NFT parser (which calls `parseDigitalAsset(raw, include?.collection)`).

**`buildCreatorWhere(filter)`** — translate CreatorFilter to Hasura `lsp4_creator_bool_exp`.

Filter → Hasura mapping (all 7 from CONTEXT.md):

- `filter.creatorAddress` → `{ creator_address: { _ilike: \`%${escapeLike(creatorAddress)}%\` } }` — partial match for filtering
- `filter.digitalAssetAddress` → `{ address: { _ilike: \`%${escapeLike(digitalAssetAddress)}%\` } }`— NOTE: Hasura field is`address`, not `digital_asset_address`
- `filter.interfaceId` → `{ interface_id: { _ilike: \`%${escapeLike(interfaceId)}%\` } }` — partial match
- `filter.creatorName` → `{ creatorProfile: { lsp3Profile: { name: { _ilike: \`%${escapeLike(creatorName)}%\` } } } }` — nested profile name search
- `filter.digitalAssetName` → `{ digitalAsset: { lsp4TokenName: { value: { _ilike: \`%${escapeLike(digitalAssetName)}%\` } } } }`— nested DA name search. NOTE: Uses`lsp4TokenName.value`not`lsp4TokenName.name` — check the digital_asset_bool_exp structure in the DA service for the correct path.
- `filter.timestampFrom` → `{ timestamp: { _gte: timestampFrom } }` — ISO timestamp lower bound
- `filter.timestampTo` → `{ timestamp: { _lte: timestampTo } }` — ISO timestamp upper bound

Multiple conditions combine with `_and`. Empty filter returns empty object.

**IMPORTANT:** Check `packages/node/src/services/digital-assets.ts` `buildDigitalAssetWhere` for the exact nested DA name filter path. The DA service uses `lsp4TokenName: { value: { _ilike: ... } }` — the nested filter from lsp4_creator should use `digitalAsset: { lsp4TokenName: { value: { _ilike: ... } } }`. Also check `packages/node/src/services/followers.ts` for the profile name filter path (`creatorProfile: { lsp3Profile: { name: { _ilike: ... } } }`).

**`buildCreatorOrderBy(sort)`** — translate CreatorSort to Hasura `lsp4_creator_order_by` array.

Use `orderDir()` helper from `./utils` for all sorts. Default sort: timestamp descending (per CONTEXT.md).

**`buildCreatorIncludeVars(include)`** — translate CreatorInclude to GraphQL boolean variables.

**Export this function** — may be needed by future domains.

**Inverted default pattern:**

- When `include` is undefined → return `{}` (GraphQL defaults all Boolean! = true)
- When `include` is provided → each field defaults to `false` unless explicitly set

```typescript
export function buildCreatorIncludeVars(include?: CreatorInclude): Record<string, boolean> {
  if (!include) return {};

  const activeCreatorProfile = hasActiveIncludes(include.creatorProfile);
  const activeDigitalAsset = hasActiveIncludes(include.digitalAsset);

  const vars: Record<string, boolean> = {
    includeArrayIndex: include.arrayIndex ?? false,
    includeInterfaceId: include.interfaceId ?? false,
    includeTimestamp: include.timestamp ?? false,
    includeCreatorProfile: activeCreatorProfile,
    includeDigitalAsset: activeDigitalAsset,
  };

  // Creator profile sub-includes: reuse profile include builder with "CreatorProfile" prefix
  if (activeCreatorProfile) {
    const profileVars = buildProfileIncludeVars(include.creatorProfile);
    for (const [key, val] of Object.entries(profileVars)) {
      // includeProfileName → includeCreatorProfileName
      // includeProfileDescription → includeCreatorProfileDescription, etc.
      vars[key.replace('includeProfile', 'includeCreatorProfile')] = val;
    }
  }

  // Digital asset sub-includes: reuse DA include builder with "DigitalAsset" prefix
  if (activeDigitalAsset) {
    const daVars = buildDigitalAssetIncludeVars(include.digitalAsset);
    for (const [key, val] of Object.entries(daVars)) {
      // includeName → includeDigitalAssetName
      // includeSymbol → includeDigitalAssetSymbol, etc.
      vars[key.replace('include', 'includeDigitalAsset')] = val;
    }
  }

  return vars;
}
```

This follows the exact pattern from followers service (profile prefix replacement) combined with NFTs service (DA/collection prefix replacement). Check `packages/node/src/services/nfts.ts` for the DA prefix replacement pattern used for collection includes.

```typescript
    offset?: number;

export async function fetchCreators(
  url: string,

    filter?: CreatorFilter;
    sort?: CreatorSort;
    limit?: number;
    offset?: number;
    include?: CreatorInclude;
  },
): Promise<FetchCreatorsResult<PartialCreator>>;
export async function fetchCreators(
  url: string,
  params: {
    filter?: CreatorFilter;
    sort?: CreatorSort;
    limit?: number;
    offset?: number;
    include?: CreatorInclude;
  },
): Promise<FetchCreatorsResult | FetchCreatorsResult<PartialCreator>> {
  const where = buildCreatorWhere(params.filter);
  const orderBy = buildCreatorOrderBy(params.sort);
  const includeVars = buildCreatorIncludeVars(params.include);

  const result = await execute(url, GetCreatorsDocument, {
    where: Object.keys(where).length > 0 ? where : undefined,
    order_by: orderBy,
    limit: params.limit,
    offset: params.offset,
    ...includeVars,
  });

  if (params.include) {
    return {
      creators: parseCreators(result.lsp4_creator, params.include),
      totalCount: result.lsp4_creator_aggregate?.aggregate?.count ?? 0,
    };
  }
  return {
    creators: parseCreators(result.lsp4_creator),
    totalCount: result.lsp4_creator_aggregate?.aggregate?.count ?? 0,
  };
}
```
